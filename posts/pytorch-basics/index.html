      from torch.utils.data import Dataset
  class CustomDataset(Dataset):
      def __init__(self, text, labels):
          self.labels = labels
          self.data = text
        
      def __len__(self):
          return len(self.labels)
        
      def __getitem__(self, idx):
          label = self.labels[idx]
          text = self.data[idx]
          sample = {"Text": text, "Class": label}
          return sample
